29    public static String draw(final Lawn lawn, final Mower... mowers) {
30      int width = lawn.getWidth(), height = lawn.getHeight();
31      boolean withMower = mowers.length > 0;
32      GridBuilder builder = new GridBuilder();
33      builder.append("\n\n");
34      for (int y = height - 1; y >= 0; y--) {
35        builder.cell(y);
36        for (int x = 0; x < width; x++) {
37          Cell cell = lawn.cellAt(x, y);
38  
39          if (cell.isLock()) {
40            builder.cell(withMower ? orientation(cell, mowers) : 'x');
41          } else if (cell.isMowed()) {
42            builder.cell('-');
43          } else {
44            builder.cell(' ');
45          }
46        }
47        builder.line(width);
48      }
49  
50      builder.cell();
51      for (int i = 0; i < width; i++) {
52        builder.cell(i);
53      }
54      builder.append("\n\n");
55      return builder.toString();
56    }
57  
58    private static char orientation(final Cell cell, final Mower... mowers) {
59      return Iterables.find(Arrays.asList(mowers), new Predicate<Mower>() {
60        @Override
61        public boolean apply(final Mower input) {
62          return input.getCell().equals(cell);
63        }
64      }).getOrientation().getCode().charAt(0);
65    }

77        return (GridBuilder) this.append('\n').append(Strings.repeat(SEPARATOR_HOR, width * 5))
