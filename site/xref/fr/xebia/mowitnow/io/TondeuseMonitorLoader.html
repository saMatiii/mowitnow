33    }

36      return fromLines(Splitter.on(LINE_SEPRATOR).omitEmptyStrings().splitToList(text));
37    }

39    public TondeuseMoniteur fromLines(@NonNull final List<String> pLines) {
40      Preconditions.checkArgument(pLines.size() > 3,
41          "Erreur de chargement du moniteur: ficher incomplet");
42      Iterator<String> lines = pLines.iterator();
43      PelouseParseur pelouseParser = new PelouseParseur();
44      InstructionParseur instructionParser = new InstructionParseur();
45      Pelouse pelouse = pelouseParser.parse(lines.next());
46      TondeuseParseur tondeuseParser = new TondeuseParseur(pelouse);
47      List<Tondeuse> tondeuses = Lists.newArrayList();
48      while (lines.hasNext()) {
49        Tondeuse tondeuse = tondeuseParser.parse(lines.next());
50        Queue<Instruction> instructions = instructionParser.parse(lines.next());
51        tondeuse.setInstructions(instructions);
52        tondeuses.add(tondeuse);
53      }
54      return new TondeuseMoniteur(pelouse, tondeuses);
55    }
