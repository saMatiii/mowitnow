43    private void lawn(final int width, final int height) {
44      // When
45      Lawn lawn = new Lawn(width, height);
46  
47      // Then
48      for (int x = 0; x < width; x++) {
49        for (int y = 0; y < height; y++) {
50          Cell cell = lawn.cellAt(x, y);
51          assertEquals(Position.at(x, y), cell.getPosition());
52          assertFalse(cell.isLock());
53          assertFalse(cell.isMowed());
54          assertNext(lawn, EAST, cell.getPosition(), at(x + 1, y));
55          assertNext(lawn, WEST, cell.getPosition(), at(x - 1, y));
56          assertNext(lawn, NORTH, cell.getPosition(), at(x, y + 1));
57          assertNext(lawn, SOUTH, cell.getPosition(), at(x, y - 1));
58        }
59      }
60    }
61  
62    private void assertNext(final Lawn lawn, final Orientation orientation, final Position position,
63        final Position expectedPosition) {
64      Cell cell = lawn.cellAt(position);
65      Cell actual = cell.next(orientation);
66      Cell expected = null;
67      if (in(expectedPosition.getX(), 0, lawn.getWidth())
68          && in(expectedPosition.getY(), 0, lawn.getHeight())) {
69        expected = lawn.cellAt(expectedPosition);
70      }
71      assertEquals(expected, actual);
72    }
73  
74    private boolean in(final int position, final int min, final int max) {
75      return position >= min && position < max;
76    }
